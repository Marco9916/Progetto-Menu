#include <LiquidCrystal.h>
LiquidCrystal lcd(12 , 11 , 5 , 4 , 3 , 2);
//VARIABILI
const int avanti = 9;
const int invio = 8;
const int indietro = 6;
const int led = 10;
int R_Avanti=0;
int R_Invio = 0;
int R_Indietro = 0;
int R_Led = 0;
int R_Avanti_numero = 1;
int R_Invio_numero = 0;
int R_Indietro_numero = 0;

void setup() {

  Serial.begin(9600);
  lcd.begin(16 ,2);
  pinMode(avanti , INPUT);
  pinMode(invio , INPUT);
  pinMode(indietro , INPUT);
  pinMode(led , OUTPUT);
  lcd.setCursor(0,0);
  lcd.print("   BENVENUTO!");
  delay(2000);
  lcd.setCursor(0,1);
  lcd.print("    ATTENDI");
  delay(2000);
  lcd.clear();

}

void loop() {
  R_Led = digitalRead(led);
  R_Avanti = digitalRead(avanti);
  R_Invio = digitalRead(invio);
  R_Indietro = digitalRead(indietro);
  if(R_Avanti == HIGH){
    R_Avanti_numero +=1;
  }
    if(R_Indietro == HIGH){
    R_Avanti_numero = R_Avanti_numero-1;
  }
  switch(R_Avanti_numero){
    //CASE 1 CON 1 FUNZIONE
    case 1:
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("  Temperatura.");
    if(R_Invio == 1){
      R_Invio_numero += 1;  
      if(R_Invio_numero >1){R_Invio_numero=0;}
    }

    switch(R_Invio_numero){
        case 1:
        lcd.clear();
        lcd.print("Temperatura 00.0");
      }
    break;

    //CASE 2 CON 1 FUNZIONE
    case 2:
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("Seconda Scelta");
    if(R_Invio == 1){
      R_Invio_numero += 1;  
      if(R_Invio_numero >1){R_Invio_numero=0;}
    }

    switch(R_Invio_numero){
        case 1:
        lcd.clear();
        lcd.print("SECONDA FUNZIONE");
      }
    break;

    //CASE 3 CON 1 FUNZIONE
    case 3:
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("Terza Scelta");
    if(R_Invio == 1){
      R_Invio_numero += 1;  
      if(R_Invio_numero >1){R_Invio_numero=0;}
    }

    switch(R_Invio_numero){
        case 1:
        lcd.clear();
        lcd.print("TERZA FUNZIONE");
      }
    break;

    //CASE 4   CON 1 FUNZIONE
    case 4:
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("Quarta Scelta");
    if(R_Invio == 1){
      R_Invio_numero += 1;  
      if(R_Invio_numero >1){R_Invio_numero=0;}
    }

    switch(R_Invio_numero){
        case 1:
        lcd.clear();
        lcd.print("QUARTA FUNZIONE");
      }
    break;

    //CASE 5 CON 1 FUNZIONE
    case 5:
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("Quinta Scelta");
    if(R_Invio == 1){
      R_Invio_numero += 1;  
      if(R_Invio_numero >1){R_Invio_numero=0;}
    }

    switch(R_Invio_numero){
        case 1:
        lcd.clear();
        lcd.print("QUINTA FUNZIONE");
      }
      break;
    }

    if((R_Avanti == HIGH) || (R_Invio == HIGH) || (R_Indietro == HIGH)){digitalWrite(led , HIGH);}
    else {digitalWrite(led , LOW  );}

    if(R_Avanti_numero>5){R_Avanti_numero=1;}
    else if(R_Avanti_numero<1){R_Avanti_numero=5;}
    else{R_Avanti_numero=R_Avanti_numero;}


    Serial.print(R_Avanti_numero);
    delay(500);
    Serial.print(R_Invio_numero);
    delay(500);
}
